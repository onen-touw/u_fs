338 static ufs_type_t get_fs_type(const fs::path& path) {
339 std::string str = path.string();
340
341 if (str.find("/spiffs") == 0) return ufs_type_t::spiffs;
342 if (str.find("/littlefs") == 0) return ufs_type_t::littlefs;
343 if (str.find("/sdcard") == 0) return ufs_type_t::fat;
344 if (str.find("/sdmmc") == 0) return ufs_type_t::fat;
